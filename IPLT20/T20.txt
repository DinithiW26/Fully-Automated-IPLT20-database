Database

mysql> CREATE DATABASE t20ipl2014;

mysql> USE t20ipl2014;

mysql> CREATE TABLE Team
(
TeamID INT(10) NOT NULL AUTO_INCREMENT PRIMARY KEY,
TeamName VARCHAR(60) NOT NULL
);

mysql> CREATE TABLE Bowler
(
BowlerID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
BowlerName VARCHAR(60) NOT NULL,
TeamID INT(2) NOT NULL,
FOREIGN KEY(TeamID) REFERENCES Team (TeamID)
);

mysql> CREATE TABLE Batsman
(
BatsmanID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
BatsmanName VARCHAR(60) NOT NULL,
TeamID INT(2) NOT NULL,
FOREIGN KEY(TeamID) REFERENCES Team (TeamID)
);

mysql> CREATE TABLE Matches
(
MatchNumber INT(3) NOT NULL PRIMARY KEY,
MatchDecription VARCHAR(30) NOT NULL,
Venue VARCHAR(30) NOT NULL,
MatchDate VARCHAR(20) NOT NULL,
BatFirstTeamID INT(2) NOT NULL,
BatSecondTeamID INT(2) NOT NULL,
TeamWonID INT(2),
MatchResult VARCHAR(100) NOT NULL,
FOREIGN KEY(BatFirstTeamID) REFERENCES Team (TeamID),
FOREIGN KEY(BatSecondTeamID) REFERENCES Team (TeamID),
FOREIGN KEY(TeamWonID) REFERENCES Team (TeamID)
);

mysql> CREATE TABLE CommentaryData
(
BallIndex INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
MatchNumber INT(3) NOT NULL,
Innings INT(1) NOT NULL,
Over FLOAT(3,1) NOT NULL,
BowlerID INT NOT NULL,
FOREIGN KEY(BowlerID) REFERENCES Bowler(BowlerID),
BatsmanID INT NOT NULL,
FOREIGN KEY(BatsmanID) REFERENCES Batsman(BatsmanID),
Runs INT(4),
Wicket BOOLEAN,
DismissalType VARCHAR(20),
NOBall BOOLEAN,
Wide BOOLEAN,
LBBye BOOLEAN,
Four BOOLEAN,
Six BOOLEAN,
Score INT
);
